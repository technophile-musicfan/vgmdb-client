**Selector reference (from the captured fixtures — refine empirically during TDD):**
- Title: `h1 span.albumtitle[lang="en"|"ja"|"ja-Latn"]`. Placeholder spans are `display:none` and prefixed `<em> / </em>`; text after the `/` may duplicate English.
- Info box: `table#album_infobit_large tr` → `td span.label b` is the field name, the sibling `td` is the value. Fields seen: "Catalog Number", "Release Date", "Classification". Release Date value is an `<a>` whose href ends `#YYYYMMDD` (robust date source) and whose text is e.g. "Nov 08, 2005".
- Covers: `medium-media.vgm.io/...` (full) and `thumb-media.vgm.io/...` (small) image URLs.
- Tracklist: `#tracklist span.tl` — one per language tab (`id="tl<n>"`). Each contains `Disc N` headers and `table.role tr.rolebit` rows: `td span.label` = track number, the wide `td` = title text, `td span.time` = length.
- Credits: `tr.maincred` rows (label + artist links `a[href^="/artist/"]` with `span.artistname[lang]`).

```python
from __future__ import annotations

import lxml.html
from lxml.html import HtmlElement


def parse_tree(html: str) -> HtmlElement:
    """Parse an HTML document into an lxml element tree."""
    return lxml.html.fromstring(html)


def text(element: HtmlElement | None) -> str:
    """Whitespace-collapsed text content of an element (entities already decoded by lxml)."""
    if element is None:
        return ""
    return " ".join(element.text_content().split())
```

```python
import re

from vgmdb_client.models import LocalizedText

_LANG_LABELS = {"en": "English", "ja": "Japanese", "ja-Latn": "Romaji"}
_CJK = re.compile(r"[぀-ヿ㐀-䶿一-鿿＀-￯]")


def _span_text(span: HtmlElement) -> str:
    # Placeholder spans render a leading "<em> / </em>"; the <em> is dropped by text_content,
    # leaving a leading "/ " — strip a single leading slash separator.
    raw = " ".join(span.text_content().split())
    return re.sub(r"^/\s*", "", raw).strip()


def localized_text(spans: list[HtmlElement]) -> LocalizedText:
    """Build LocalizedText from language spans, applying the placeholder rule.

    en -> English; ja -> Japanese only when it contains real Japanese script;
    ja-Latn -> Romaji only when it differs from the English text. Duplicates dropped.
    """
    by_lang = {span.get("lang"): _span_text(span) for span in spans if span.get("lang")}
    english = by_lang.get("en", "")
    out: dict[str, str] = {}
    if english:
        out["English"] = english
    ja = by_lang.get("ja")
    if ja and _CJK.search(ja):
        out["Japanese"] = ja
    romaji = by_lang.get("ja-Latn")
    if romaji and romaji != english:
        out["Romaji"] = romaji
    return LocalizedText(out)
```

```python
from vgmdb_client.models import PartialDate

_BASE = "https://vgmdb.net"
_DATE_FRAGMENT = re.compile(r"#(\d{4})(\d{2})?(\d{2})?$")


def partial_date(href_or_text: str | None) -> PartialDate | None:
    """Parse a vgmdb date: prefer the #YYYYMMDD calendar fragment, else PartialDate.parse."""
    if not href_or_text:
        return None
    match = _DATE_FRAGMENT.search(href_or_text)
    if match:
        year = int(match.group(1))
        month = int(match.group(2)) if match.group(2) else None
        day = int(match.group(3)) if match.group(3) else None
        try:
            return PartialDate(year=year, month=month, day=day)
        except ValueError:
            return None
    return PartialDate.parse(href_or_text)


def absolute_url(href: str | None) -> str | None:
    """Make a vgmdb-relative href absolute; pass through absolute URLs; None -> None."""
    if not href:
        return None
    if href.startswith("http://") or href.startswith("https://"):
        return href
    return _BASE + href if href.startswith("/") else f"{_BASE}/{href}"
```
